import { Meta, Story, Props } from '@storybook/addon-docs';

import {
	ICON_BUTTON_SIZES,
	ICON_BUTTON_COLOR_SCHEMES,
	ICON_BUTTON_TYPES,
	ICON_BUTTON_COLORS,
} from 'design-system/lib/js/components/Buttons/IconButton/IconButton.consts';
import {
	BUTTON_ELEVATIONS,
	BUTTON_RADIUSES,
} from 'design-system/lib/js/components/Buttons/Button/Button.consts';
import { ICONS } from 'design-system/lib/js/components/Icon/Icon.consts';

<Meta title="components/Buttons/IconButton" component={WnlIconButton} />

# IconButton

<Story
	name="interactive"
	args={{
		size: ICON_BUTTON_SIZES.LARGE,
		type: ICON_BUTTON_TYPES.ICON_ONLY,
		radius: BUTTON_RADIUSES.CAPSULE,
		label: 'Zamknij',
		colorScheme: ICON_BUTTON_COLOR_SCHEMES.ALL_IN_COLOR,
		touchable: true,
		elevation: BUTTON_ELEVATIONS.NONE,
		color: 'PRIMARY',
	}}
	argTypes={{
		size: {
			control: { type: 'select', options: Object.values(ICON_BUTTON_SIZES) },
			defaultValue: ICON_BUTTON_SIZES.LARGE,
		},
		type: {
			control: { type: 'select', options: Object.values(ICON_BUTTON_TYPES) },
			defaultValue: ICON_BUTTON_TYPES.ICON_ONLY,
		},
		radius: {
			control: { type: 'select', options: Object.values(BUTTON_RADIUSES) },
			defaultValue: BUTTON_RADIUSES.CAPSULE,
		},
		icon: {
			control: { type: 'select', options: Object.keys(ICONS) },
			defaultValue: 'FA_TIMES_LIGHT',
		},
		color: {
			control: { type: 'select', options: Object.keys(ICON_BUTTON_COLORS) },
			defaultValue: 'PRIMARY',
		},
		colorScheme: {
			control: { type: 'select', options: Object.values(ICON_BUTTON_COLOR_SCHEMES) },
			defaultValue: ICON_BUTTON_COLOR_SCHEMES.ALL_IN_COLOR,
		},
		elevation: {
			control: { type: 'select', options: Object.values(BUTTON_ELEVATIONS) },
			defaultValue: BUTTON_ELEVATIONS.NONE,
		},
	}}
>
	{(args) => {
		return {
			components: { WnlIconButton },
			props: Object.keys(args),
			template: `<wnl-icon-button
              :icon="ICONS[icon]"
              :size="size"
              :type="type"
              :radius="radius"
              :color-scheme="colorScheme"
              :touchable="touchable"
              :elevation="elevation"
              :color="ICON_BUTTON_COLORS[color]"
            ><span v-if="label !== ''">{{label}}</span></wnl-icon-button>`,
			created() {
				this.ICONS = ICONS;
				this.ICON_BUTTON_COLORS = ICON_BUTTON_COLORS;
			},
		};
	}}
</Story>
