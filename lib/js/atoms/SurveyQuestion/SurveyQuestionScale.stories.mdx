		   disabled: { control: { type: 'boolean' }, defaultValue: false },

					{ id: '1', label: 'Nie zgadzam się', color: SURVEY_TOGGLE_COLORS.PRIMARY, content: '1'},

					{ id: '4', label: 'Zgadzam się ', color: SURVEY_TOGGLE_COLORS.PRIMARY, content: '4'},
				];
			},
			template: '<survey-question-scale :title="title" :options="options" :elaborationLabel="elaborationLabel" :placeholder="placeholder" :selected="selected" :disabled="disabled" @select-change="selected = $event">' +
				'<div v-if="explanation" slot="explanation">{{ explanation }}</div>' +
				'</survey-question-scale>',
		};
	}}
</Story>
