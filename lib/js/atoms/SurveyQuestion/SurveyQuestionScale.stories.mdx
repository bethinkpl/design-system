# SurveyQuestionScale

<Story name='interactive'
	   args={{
		  title: 'title',
		   elaborationLabel: 'elaborationLabel',
		   disabled: false,
	   }}
	   argTypes={{
		   title: { control: { type: 'text' } },
		   elaborationLabel: { control: { type: 'text' } },
		   explanation: { control: { type: 'text' }, defaultValue: 'Wyczerpujące materiały dają poczucie pełnego zrozumieina, przy jednoczesnym usystematyzowaniu informacji. Wpływa na to nie tylko ich jakość, ale też ilość.' },
		   disabled: { control: { type: 'boolean' } },
	   }}>
	{(args) => {
		return {
			components: { SurveyQuestionScale },
			props: Object.keys(args),
			data() {
				return { selected: null }
			},
			created() {
				this.options = [
					{ id: '1', label: 'Nie zgadzam się', color: SURVEY_TOGGLE_COLORS.PRIMARY, content: '1'},
					{ id: '2', label: 'Trochę się nie zgadzam', color: SURVEY_TOGGLE_COLORS.PRIMARY, content: '2'},
					{ id: '3', label: 'Trochę się zgadzam', color: SURVEY_TOGGLE_COLORS.PRIMARY, content: '3'},
					{ id: '4', label: 'Zgadzam się ', color: SURVEY_TOGGLE_COLORS.PRIMARY, content: '4'},
				];
			},
			template: '<survey-question-scale :title="title" :options="options" :elaborationLabel="elaborationLabel" :status="status" :selected="selected" :disabled="disabled" @selectChange="selected = $event">' +
				'<div slot="explanation">{{ explanation }}</div>' +
				'</survey-question-scale>',
		};
	}}
</Story>
